```python
# ai_client.py - 行业检测核心逻辑
def detect_industry(name: str) -> tuple:
    name_lower = name.lower()
    
    if any(k in name_lower for k in ["银行", "保险", "平安"]):
        return ("金融服务业", ..., [...], [...])
        
    elif any(k in name_lower for k in ["新能源", "储能", "思格"]):
        return ("新能源与储能行业", ..., [...], [...])
        
    elif any(k in name_lower for k in ["制造", "万向"]):
        return ("高端制造业", ..., [...], [...])
    
    # ... 其他行业
    
    return ("现代服务业", ..., [...], [...])

# 使用 MD5 hash 保证一致性
hash_val = int(hashlib.md5(company_name.encode()).hexdigest()[:8], 16)
random.seed(hash_val)  # 同一公司每次结果相同
```

```typescript
      const trendData = trendResponse?.data?.data || trendResponse?.data;
      
      if (scoreData && trendData) {
        setEmotionScore(scoreData);
        setEmotionTrend(trendData);
      } else {
        setEmotionScore(null);
        setEmotionTrend(null);
      }
    } catch (err) {
      console.error('获取情绪数据失败:', err);
      setEmotionScore(null);  // 明确清空状态
      setEmotionTrend(null);
    }
  };
  
  return (
    <div>
      {/* 大号分数显示 */}
      <div className="text-6xl font-bold">
        {getEmotionIcon(emotionScore?.current_label)}
        {emotionScore?.current_score || '--'} 分
      </div>
      
      {/* 三列指标卡片 */}
      <div className="grid grid-cols-3 gap-4">
        <MetricCard title="当前" value={emotionScore?.current_score} />
        <MetricCard title="7日平均" value={emotionScore?.last_7d_avg} />
        <MetricCard title="30日平均" value={emotionScore?.last_30d_avg} />
      </div>
      
      {/* 趋势折线图 */}
      {emotionTrend && (
        <Line data={formatChartData(emotionTrend.trend)} />
      )}
    </div>
  );
};
```

### 行业对标数据流
```
[输入公司名: "思格新能源"]
      │
      ▼
[行业检测] (detect_industry())
      │
      ├── 包含 "新能源"/"储能"/"思格"
      │
      ▼
[确定行业: "新能源与储能行业"]
      │
      ▼
[Hash计算] (MD5(company_name))
      │
      ▼
[固定随机种子] (random.seed(hash_val))
      │
      ▼
[生成差异化数据]
  ├── market_share: "12%"
  ├── revenue_growth: "14%"
  ├── peer1: "宁德时代" (真实公司)
  ├── peer2: "比亚迪" (真实公司)
  └── strengths: ["技术壁垒高", ...]
      │
      ▼
[返回JSON] → [前端渲染SWOT图表]
```

### 新闻聚合数据流
```
[用户请求新闻]
      │
      ▼
[查询数据库]
      │
      ├── 有数据 → 直接返回
      │
      └── 无数据 → [自动生成]
            │
            ▼
      [检测行业] (8大行业模板)
            │
```
